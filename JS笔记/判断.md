# 判断

## if判断

- if的基本语法

  ```js
  /*if(条件){
    执行代码 
  }*/
  if(4>3){
      alert("条件为真执行代码");
  	//条件为假，则不执行
  }else{
      alert("条件为假执行代码");
      //条件为假时执行的代码
  }
  //一但条件满足，只执行if里面的代码，后面的else里的代码就直接忽略不执行了
  var a =100;
  if(a<50){
      console.log(1);
  }else if(a<70){
      console.log(2);
  }else if(a<90){
      console.log(3);
  }else if(a<110){
      console.log(4);
  }else if(a<130){
      console.log(5);
  }
  if(true){
      
  }else{
      Alert(1);
  	//else直接被忽略不执行
  }
  ```

  ```html
  <!DOCTYPE html>
  <html lang="en">
  <head>
      <meta charset="UTF-8">
      <title>课堂案例</title>
      <style>
          *{
              margin: 0;
          }
          .wrap{
              margin: 50px 0 0 50px;
              width: 800px;
              height: 210px;
          }
          table{
              width: 100%;
              border-collapse: collapse;
          }
          th{
              height: 50px;
              border: 1px solid black;
          }
          td{
              border: 1px solid black;
              text-align: center;
          }
          .img{
              width: 25%;
              text-align: center;
          }
          span{
              display: inline-block;
              width: 60px;
              height: 60px;
              text-align: center;
              line-height: 60px;
              font-size: 20px;
          }
          #plus,#reduce{
              background-color: aqua;
              cursor: pointer;
              user-select: none;
          }
      </style>
  </head>
  <body>
      <div class="wrap">
          <table>
              <tr>
                  <th>商品</th>
                  <th>单价</th>
                  <th>数量</th>
                  <th>价格</th>
              </tr>
              <tr>
                  <td class="img"><img src="./1.jpg" alt=""></td>
                  <td>
                      <span>20</span>
                  </td>
                  <td>
                      <span id="reduce">-</span>
                      <span id="number">0</span>
                      <span id="plus">+</span>
                  </td>
                  <td>
                      <span id="total">0</span>
                  </td>
  
              </tr>
          </table>
      </div>
      <script>
          /*1、获取需要操作or改变的元素*/
          var oPlus = document.getElementById("plus");
          var oNumber = document.getElementById("number");
          var oReduce = document.getElementById("reduce");
          var oTotal = document.getElementById("total");
          //点击+号商品数量增加
          oPlus.onclick = function () {
              /*
              2、获取当前Number的内容是多少
              注意：获取到的内容先减去0让字符串变数字
              */
              var number = oNumber.innerHTML - 0;
              number ++;
              /*5、计算价格*/
              oTotal.innerHTML = number*20;
              /*
              3、赋值 oNumber.innerHTML = ++number;
              注意：因为后置++先赋值，后自增，所以在这里要使用前置++，先自增再赋值
              */
              oNumber.innerHTML = number;
              
          }
          //点击减号减少商品数量
          oReduce.onclick = function () {
              var number = oNumber.innerHTML - 0;
              number --;
              /*
              4、为了不产生负数，赋值之前要判断 number 不能小于0用作限制
  			if(number>0){
  			   number = 0;
              }
              oNumber.innerHTML = number;
              */
              if(number>=0){
                  oTotal.innerHTML = number * 20;
                  oNumber.innerHTML = number;
              }
          }
      </script>
  </body>
  </html>
  ```
